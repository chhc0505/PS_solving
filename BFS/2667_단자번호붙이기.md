## Problem
<https://www.acmicpc.net/problem/2667>

BFS

## Comment
* 시간복잡도 : O(${n^2+4*n^2+sorting}$)

* 공간복잡도 : 2500byte+a

* BFS floodfill 기본문제!

## Code
```c++
#include<bits/stdc++.h>
#define X first
#define Y second
using namespace std;

int n,ans(0);
int dx[4]={0,1,0,-1};
int dy[4]={1,0,-1,0};

vector<int> cntArr;
queue<pair<int,int> > q;

string board[26];
bool visited[26][26];

void bfs(int x, int y) {
    visited[x][y]=true;
    q.push({x,y});
    int cnt(0);
    cnt++;
    
    while (!q.empty()) {
        auto curr=q.front(); q.pop();
        for (int dir=0; dir<4; dir++) {
            int nx=curr.X+dx[dir];
            int ny=curr.Y+dy[dir];
            if (nx<0 || nx>=n || ny<0 || ny>=n) continue;
            if (board[nx][ny]=='0' || visited[nx][ny]) continue;
            visited[nx][ny]=true;
            q.push({nx,ny});
            cnt++;
        }
    }
    
    cntArr.push_back(cnt);
    return;
}
int main () {
    
    ios::sync_with_stdio(0); cin.tie(0);
    cin >> n;
    
    for (int i=0; i<n; i++) {
        cin >> board[i];
    }
    
    for (int i=0; i<n; i++) {
        for (int j=0; j<n; j++) {
            if (board[i][j]!='0' && visited[i][j]==false) {
                ans++;
                bfs(i,j);
            }
        }
    }
    
    sort(cntArr.begin(),cntArr.end());
    
    cout << ans << '\n';
    for (auto c : cntArr) {
        cout << c << '\n';
    }
    
    return 0;
}
```
